Dataset:

Questions:

1-Which year has the highest release of movies?
2-Which genres are most popular from year to year?
3-What kinds of properties are associated with movies that have high revenues?

import matplotlib.pyplot as plt
import seaborn as sns

def count_genre(x): # make function to split string and count gente then return it.
    data_plot=reader[x].str.cat(sep ='|') #concatenates the row of genre.
    data=pd.Series(data_plot.split('|'))
    info=data.value_counts(ascending=False)# count the genre and return it.
    return info


total_genre_movies= count_genre('genres')
total_genre_movies.plot(kind= 'barh', figsize=(13,6),fontsize=12,colormap='tab20c')

plt.title("Genre With Highest Release",fontsize=15) #set up plot.
plt.xlabel('Number of movies',fontsize=13)
plt.ylabel('Genres',fontsize=13)
sns.set_style("whitegrid")
